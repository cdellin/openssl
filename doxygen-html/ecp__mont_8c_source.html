198 int ec_GFp_mont_group_set_curve(EC_GROUP *group, const BIGNUM *p, const BIGNUM *a, const BIGNUM *b, BN_CTX *ctx)
199  {
200  BN_CTX *new_ctx = NULL;
201  BN_MONT_CTX *mont = NULL;
202  BIGNUM *one = NULL;
203  int ret = 0;
204 
205  if (group->field_data1 != NULL)
206  {
207  BN_MONT_CTX_free(group->field_data1);
208  group->field_data1 = NULL;
209  }
210  if (group->field_data2 != NULL)
211  {
212  BN_free(group->field_data2);
213  group->field_data2 = NULL;
214  }
215 
216  if (ctx == NULL)
217  {
218  ctx = new_ctx = BN_CTX_new();
219  if (ctx == NULL)
220  return 0;
221  }
222 
223  mont = BN_MONT_CTX_new();
224  if (mont == NULL) goto err;
225  if (!BN_MONT_CTX_set(mont, p, ctx))
226  {
227  ECerr(EC_F_EC_GFP_MONT_GROUP_SET_CURVE, ERR_R_BN_LIB);
228  goto err;
229  }
230  one = BN_new();
231  if (one == NULL) goto err;
232  if (!BN_to_montgomery(one, BN_value_one(), mont, ctx)) goto err;
233 
234  group->field_data1 = mont;
235  mont = NULL;
236  group->field_data2 = one;
237  one = NULL;
238 
239  ret = ec_GFp_simple_group_set_curve(group, p, a, b, ctx);
240 
241  if (!ret)
242  {
243  BN_MONT_CTX_free(group->field_data1);
244  group->field_data1 = NULL;
245  BN_free(group->field_data2);
246  group->field_data2 = NULL;
247  }
248 
249  err:
250  if (new_ctx != NULL)
251  BN_CTX_free(new_ctx);
252  if (mont != NULL)
253  BN_MONT_CTX_free(mont);
254  return ret;
255  }
