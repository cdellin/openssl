70 int PEM_SealInit(PEM_ENCODE_SEAL_CTX *ctx, EVP_CIPHER *type, EVP_MD *md_type,
71  unsigned char **ek, int *ekl, unsigned char *iv, EVP_PKEY **pubk,
72  int npubk)
73  {
74  unsigned char key[EVP_MAX_KEY_LENGTH];
75  int ret= -1;
76  int i,j,max=0;
77  char *s=NULL;
78 
79  for (i=0; i<npubk; i++)
80  {
81  if (pubk[i]->type != EVP_PKEY_RSA)
82  {
83  PEMerr(PEM_F_PEM_SEALINIT,PEM_R_PUBLIC_KEY_NO_RSA);
84  goto err;
85  }
86  j=RSA_size(pubk[i]->pkey.rsa);
87  if (j > max) max=j;
88  }
89  s=(char *)OPENSSL_malloc(max*2);
90  if (s == NULL)
91  {
92  PEMerr(PEM_F_PEM_SEALINIT,ERR_R_MALLOC_FAILURE);
93  goto err;
94  }
95 
96  EVP_EncodeInit(&ctx->encode);
97 
98  EVP_MD_CTX_init(&ctx->md);
99  if (!EVP_SignInit(&ctx->md,md_type))
100  goto err;
101 
102  EVP_CIPHER_CTX_init(&ctx->cipher);
103  ret=EVP_SealInit(&ctx->cipher,type,ek,ekl,iv,pubk,npubk);
104  if (ret <= 0) goto err;
105 
106  /* base64 encode the keys */
107  for (i=0; i<npubk; i++)
108  {
109  j=EVP_EncodeBlock((unsigned char *)s,ek[i],
110  RSA_size(pubk[i]->pkey.rsa));
111  ekl[i]=j;
112  memcpy(ek[i],s,j+1);
113  }
114 
115  ret=npubk;
116 err:
117  if (s != NULL) OPENSSL_free(s);
118  OPENSSL_cleanse(key,EVP_MAX_KEY_LENGTH);
119  return(ret);
120  }

122 int PEM_SealUpdate(PEM_ENCODE_SEAL_CTX *ctx, unsigned char *out, int *outl,
123  unsigned char *in, int inl)
124  {
125  unsigned char buffer[1600];
126  int i,j;
127 
128  *outl=0;
129  if (!EVP_SignUpdate(&ctx->md,in,inl))
130  return 0;
131  for (;;)
132  {
133  if (inl <= 0) break;
134  if (inl > 1200)
135  i=1200;
136  else
137  i=inl;
138  if (!EVP_EncryptUpdate(&ctx->cipher,buffer,&j,in,i))
139  return 0;
140  EVP_EncodeUpdate(&ctx->encode,out,&j,buffer,j);
141  *outl+=j;
142  out+=j;
143  in+=i;
144  inl-=i;
145  }
146  return 1;
147  }

150  unsigned char *out, int *outl, EVP_PKEY *priv)
151  {
152  unsigned char *s=NULL;

159  goto err;
160  }
161  i=RSA_size(priv->pkey.rsa);
162  if (i < 100) i=100;
163  s=(unsigned char *)OPENSSL_malloc(i*2);
164  if (s == NULL)
165  {
166  PEMerr(PEM_F_PEM_SEALFINAL,ERR_R_MALLOC_FAILURE);
167  goto err;
168  }
169 
170  if (!EVP_EncryptFinal_ex(&ctx->cipher,s,(int *)&i))
171  goto err;
172  EVP_EncodeUpdate(&ctx->encode,out,&j,s,i);
173  *outl=j;
174  out+=j;
175  EVP_EncodeFinal(&ctx->encode,out,&j);
176  *outl+=j;
177 
178  if (!EVP_SignFinal(&ctx->md,s,&i,priv)) goto err;
179  *sigl=EVP_EncodeBlock(sig,s,i);
180 
181  ret=1;
182 err:
183  EVP_MD_CTX_cleanup(&ctx->md);
184  EVP_CIPHER_CTX_cleanup(&ctx->cipher);
185  if (s != NULL) OPENSSL_free(s);
186  return(ret);
187  }
