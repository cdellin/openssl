115 static int rc4_hmac_md5_cipher(EVP_CIPHER_CTX *ctx, unsigned char *out,
116  const unsigned char *in, size_t len)
117  {
118  EVP_RC4_HMAC_MD5 *key = data(ctx);
119 #if defined(STITCHED_CALL)
120  size_t rc4_off = 32-1-(key->ks.x&(32-1)), /* 32 is $MOD from rc4_md5-x86_64.pl */
121  md5_off = MD5_CBLOCK-key->md.num,
122  blocks;
123  unsigned int l;
124  extern unsigned int OPENSSL_ia32cap_P[];
125 #endif
126  size_t plen = key->payload_length;
127 
128  if (plen!=NO_PAYLOAD_LENGTH && len!=(plen+MD5_DIGEST_LENGTH)) return 0;
129 
130  if (ctx->encrypt) {
131  if (plen==NO_PAYLOAD_LENGTH) plen = len;
132 #if defined(STITCHED_CALL)
133  /* cipher has to "fall behind" */
134  if (rc4_off>md5_off) md5_off+=MD5_CBLOCK;
135 
136  if (plen>md5_off && (blocks=(plen-md5_off)/MD5_CBLOCK) &&
137  (OPENSSL_ia32cap_P[0]&(1<<20))==0) {
138  MD5_Update(&key->md,in,md5_off);
139  RC4(&key->ks,rc4_off,in,out);
140 
141  rc4_md5_enc(&key->ks,in+rc4_off,out+rc4_off,
142  &key->md,in+md5_off,blocks);
143  blocks *= MD5_CBLOCK;
144  rc4_off += blocks;
145  md5_off += blocks;
146  key->md.Nh += blocks>>29;
147  key->md.Nl += blocks<<=3;
148  if (key->md.Nl<(unsigned int)blocks) key->md.Nh++;
149  } else {
150  rc4_off = 0;
151  md5_off = 0;
152  }
153 #endif
154  MD5_Update(&key->md,in+md5_off,plen-md5_off);
155 
156  if (plen!=len) { /* "TLS" mode of operation */
157  if (in!=out)
158  memcpy(out+rc4_off,in+rc4_off,plen-rc4_off);
159 
160  /* calculate HMAC and append it to payload */
161  MD5_Final(out+plen,&key->md);
162  key->md = key->tail;
163  MD5_Update(&key->md,out+plen,MD5_DIGEST_LENGTH);
164  MD5_Final(out+plen,&key->md);
165  /* encrypt HMAC at once */
166  RC4(&key->ks,len-rc4_off,out+rc4_off,out+rc4_off);
167  } else {
168  RC4(&key->ks,len-rc4_off,in+rc4_off,out+rc4_off);
169  }
170  } else {
171  unsigned char mac[MD5_DIGEST_LENGTH];
172 #if defined(STITCHED_CALL)
173  /* digest has to "fall behind" */
174  if (md5_off>rc4_off) rc4_off += 2*MD5_CBLOCK;
175  else rc4_off += MD5_CBLOCK;
176 
177  if (len>rc4_off && (blocks=(len-rc4_off)/MD5_CBLOCK) &&
178  (OPENSSL_ia32cap_P[0]&(1<<20))==0) {
179  RC4(&key->ks,rc4_off,in,out);
180  MD5_Update(&key->md,out,md5_off);
181 
182  rc4_md5_enc(&key->ks,in+rc4_off,out+rc4_off,
183  &key->md,out+md5_off,blocks);
184  blocks *= MD5_CBLOCK;
185  rc4_off += blocks;
186  md5_off += blocks;
187  l = (key->md.Nl+(blocks<<3))&0xffffffffU;
188  if (l<key->md.Nl) key->md.Nh++;
189  key->md.Nl = l;
190  key->md.Nh += blocks>>29;
191  } else {
192  md5_off=0;
193  rc4_off=0;
194  }
195 #endif
196  /* decrypt HMAC at once */
197  RC4(&key->ks,len-rc4_off,in+rc4_off,out+rc4_off);
198  if (plen!=NO_PAYLOAD_LENGTH) { /* "TLS" mode of operation */
199  MD5_Update(&key->md,out+md5_off,plen-md5_off);
200 
201  /* calculate HMAC and verify it */
202  MD5_Final(mac,&key->md);
203  key->md = key->tail;
204  MD5_Update(&key->md,mac,MD5_DIGEST_LENGTH);
205  MD5_Final(mac,&key->md);
206 
207  if (memcmp(out+plen,mac,MD5_DIGEST_LENGTH))
208  return 0;
209  } else {
210  MD5_Update(&key->md,out+md5_off,len-md5_off);
211  }
212  }
213 
214  key->payload_length = NO_PAYLOAD_LENGTH;
215 
216  return 1;
217  }

219 static int rc4_hmac_md5_ctrl(EVP_CIPHER_CTX *ctx, int type, int arg, void *ptr)
220  {
221  EVP_RC4_HMAC_MD5 *key = data(ctx);
222 
223  switch (type)
224  {
225  case EVP_CTRL_AEAD_SET_MAC_KEY:
226  {
227  unsigned int i;
228  unsigned char hmac_key[64];
229 
230  memset (hmac_key,0,sizeof(hmac_key));
231 
232  if (arg > (int)sizeof(hmac_key)) {
233  MD5_Init(&key->head);
234  MD5_Update(&key->head,ptr,arg);
235  MD5_Final(hmac_key,&key->head);
236  } else {
237  memcpy(hmac_key,ptr,arg);
238  }
239 
240  for (i=0;i<sizeof(hmac_key);i++)
241  hmac_key[i] ^= 0x36; /* ipad */
242  MD5_Init(&key->head);
243  MD5_Update(&key->head,hmac_key,sizeof(hmac_key));
244 
245  for (i=0;i<sizeof(hmac_key);i++)
246  hmac_key[i] ^= 0x36^0x5c; /* opad */
247  MD5_Init(&key->tail);
248  MD5_Update(&key->tail,hmac_key,sizeof(hmac_key));
249 
250  return 1;
251  }
252  case EVP_CTRL_AEAD_TLS1_AAD:
253  {
254  unsigned char *p=ptr;
255  unsigned int len=p[arg-2]<<8|p[arg-1];
256 
257  if (!ctx->encrypt)
258  {
259  len -= MD5_DIGEST_LENGTH;
260  p[arg-2] = len>>8;
261  p[arg-1] = len;
262  }
263  key->payload_length=len;
264  key->md = key->head;
265  MD5_Update(&key->md,p,arg);
266 
267  return MD5_DIGEST_LENGTH;
268  }
269  default:
270  return -1;
271  }
272  }
