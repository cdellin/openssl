}
else
{

if (!EC_POINT_invert(group, r, ctx)) goto err;
}

BN_CTX_free(new_ctx);
if (tmp != NULL)
EC_POINT_free(tmp);
if (wsize != NULL)
OPENSSL_free(wsize);
if (wNAF_len != NULL)

if (wNAF != NULL)

for (w = wNAF; *w != NULL; w++)

OPENSSL_free(wNAF);
}
if (val != NULL)

OPENSSL_free(val);
}
if (val_sub != NULL)
{
OPENSSL_free(val_sub);
}
return ret;
}
