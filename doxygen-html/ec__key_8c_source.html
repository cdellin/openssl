141 EC_KEY *EC_KEY_copy(EC_KEY *dest, const EC_KEY *src)
142  {
143  EC_EXTRA_DATA *d;
144 
145  if (dest == NULL || src == NULL)
146  {
147  ECerr(EC_F_EC_KEY_COPY, ERR_R_PASSED_NULL_PARAMETER);
148  return NULL;
149  }
150  /* copy the parameters */
151  if (src->group)
152  {
153  const EC_METHOD *meth = EC_GROUP_method_of(src->group);
154  /* clear the old group */
155  if (dest->group)
156  EC_GROUP_free(dest->group);
157  dest->group = EC_GROUP_new(meth);
158  if (dest->group == NULL)
159  return NULL;
160  if (!EC_GROUP_copy(dest->group, src->group))
161  return NULL;
162  }
163  /* copy the public key */
164  if (src->pub_key && src->group)
165  {
166  if (dest->pub_key)
167  EC_POINT_free(dest->pub_key);
168  dest->pub_key = EC_POINT_new(src->group);
169  if (dest->pub_key == NULL)
170  return NULL;
171  if (!EC_POINT_copy(dest->pub_key, src->pub_key))
172  return NULL;
173  }
174  /* copy the private key */
175  if (src->priv_key)
176  {
177  if (dest->priv_key == NULL)
178  {
179  dest->priv_key = BN_new();
180  if (dest->priv_key == NULL)
181  return NULL;
182  }
183  if (!BN_copy(dest->priv_key, src->priv_key))
184  return NULL;
185  }
186  /* copy method/extra data */
187  EC_EX_DATA_free_all_data(&dest->method_data);
188 
189  for (d = src->method_data; d != NULL; d = d->next)
190  {
191  void *t = d->dup_func(d->data);
192 
193  if (t == NULL)
194  return 0;
195  if (!EC_EX_DATA_set_data(&dest->method_data, t, d->dup_func, d->free_func, d->clear_free_func))
196  return 0;
197  }
198 
199  /* copy the rest */
200  dest->enc_flag = src->enc_flag;
201  dest->conv_form = src->conv_form;
202  dest->version = src->version;
203  dest->flags = src->flags;
204 
205  return dest;
206  }

260 int fips_check_ec_prng(EC_KEY *ec)
261  {
262  int bits, strength;
263  if (!FIPS_module_mode())
264  return 1;
265 
266  if (ec->flags & (EC_FLAG_NON_FIPS_ALLOW|EC_FLAG_FIPS_CHECKED))
267  return 1;
268 
269  if (!ec->group)
270  return 1;
271 
272  bits = BN_num_bits(&ec->group->order);
273 
274  if (bits < 160)
275  {
276  FIPSerr(FIPS_F_FIPS_CHECK_EC_PRNG,FIPS_R_KEY_TOO_SHORT);
277  return 0;
278  }
279  /* Comparable algorithm strengths: from SP800-57 table 2 */
280  if (bits >= 512)
281  strength = 256;
282  else if (bits >= 384)
283  strength = 192;
284  else if (bits >= 256)
285  strength = 128;
286  else if (bits >= 224)
287  strength = 112;
288  else
289  strength = 80;
290 
291 
292  if (FIPS_rand_strength() >= strength)
293  return 1;
294 
295  FIPSerr(FIPS_F_FIPS_CHECK_EC_PRNG,FIPS_R_PRNG_STRENGTH_TOO_LOW);
296  return 0;

302 int EC_KEY_generate_key(EC_KEY *eckey)
303  {
304  int ok = 0;
305  BN_CTX *ctx = NULL;
306  BIGNUM *priv_key = NULL, *order = NULL;
307  EC_POINT *pub_key = NULL;
308 
309 #ifdef OPENSSL_FIPS
310  if(FIPS_selftest_failed())
311  {
312  FIPSerr(FIPS_F_EC_KEY_GENERATE_KEY,FIPS_R_FIPS_SELFTEST_FAILED);
313  return 0;
314  }
315 #endif
316 
317  if (!eckey || !eckey->group)
318  {
319  ECerr(EC_F_EC_KEY_GENERATE_KEY, ERR_R_PASSED_NULL_PARAMETER);
320  return 0;
321  }
322 
323  if ((order = BN_new()) == NULL) goto err;
324  if ((ctx = BN_CTX_new()) == NULL) goto err;
325 
326  if (eckey->priv_key == NULL)
327  {
328  priv_key = BN_new();
329  if (priv_key == NULL)
330  goto err;
331  }
332  else
333  priv_key = eckey->priv_key;
334 
335  if (!EC_GROUP_get_order(eckey->group, order, ctx))
336  goto err;
337 
338 #ifdef OPENSSL_FIPS
339  if (!fips_check_ec_prng(eckey))
340  goto err;
341 #endif

343  do
344  if (!BN_rand_range(priv_key, order))
345  goto err;
346  while (BN_is_zero(priv_key));
347 
348  if (eckey->pub_key == NULL)
349  {
350  pub_key = EC_POINT_new(eckey->group);
351  if (pub_key == NULL)
352  goto err;
353  }
354  else

356 
357  if (!EC_POINT_mul(eckey->group, pub_key, priv_key, NULL, NULL, ctx))
358  goto err;

361  eckey->pub_key = pub_key;
362 
363 #ifdef OPENSSL_FIPS
364  if(!fips_check_ec(eckey))
365  {
366  eckey->priv_key = NULL;
367  eckey->pub_key = NULL;
368  goto err;
369  }
370 #endif
371 
372  ok=1;
373 
374 err:
375  if (order)
376  BN_free(order);
377  if (pub_key != NULL && eckey->pub_key == NULL)
378  EC_POINT_free(pub_key);
379  if (priv_key != NULL && eckey->priv_key == NULL)
380  BN_free(priv_key);
381  if (ctx != NULL)
382  BN_CTX_free(ctx);
383  return(ok);
384  }

386 int EC_KEY_check_key(const EC_KEY *eckey)
387  {
388  int ok = 0;
389  BN_CTX *ctx = NULL;
390  const BIGNUM *order = NULL;
391  EC_POINT *point = NULL;
392 
393  if (!eckey || !eckey->group || !eckey->pub_key)
394  {
395  ECerr(EC_F_EC_KEY_CHECK_KEY, ERR_R_PASSED_NULL_PARAMETER);
396  return 0;
397  }
398 
399  if (EC_POINT_is_at_infinity(eckey->group, eckey->pub_key))
400  {
401  ECerr(EC_F_EC_KEY_CHECK_KEY, EC_R_POINT_AT_INFINITY);
402  goto err;
403  }
404 
405  if ((ctx = BN_CTX_new()) == NULL)
406  goto err;
407  if ((point = EC_POINT_new(eckey->group)) == NULL)
408  goto err;
409 
410  /* testing whether the pub_key is on the elliptic curve */
411  if (!EC_POINT_is_on_curve(eckey->group, eckey->pub_key, ctx))
412  {
413  ECerr(EC_F_EC_KEY_CHECK_KEY, EC_R_POINT_IS_NOT_ON_CURVE);
414  goto err;
415  }
416  /* testing whether pub_key * order is the point at infinity */
417  order = &eckey->group->order;
418  if (BN_is_zero(order))
419  {
420  ECerr(EC_F_EC_KEY_CHECK_KEY, EC_R_INVALID_GROUP_ORDER);
421  goto err;
422  }
423  if (!EC_POINT_mul(eckey->group, point, NULL, eckey->pub_key, order, ctx))
424  {
425  ECerr(EC_F_EC_KEY_CHECK_KEY, ERR_R_EC_LIB);
426  goto err;
427  }
428  if (!EC_POINT_is_at_infinity(eckey->group, point))
429  {
430  ECerr(EC_F_EC_KEY_CHECK_KEY, EC_R_WRONG_ORDER);
431  goto err;
432  }
433  /* in case the priv_key is present :
434  * check if generator * priv_key == pub_key
435  */
436  if (eckey->priv_key)
437  {
438  if (BN_cmp(eckey->priv_key, order) >= 0)
439  {
440  ECerr(EC_F_EC_KEY_CHECK_KEY, EC_R_WRONG_ORDER);
441  goto err;
442  }
443  if (!EC_POINT_mul(eckey->group, point, eckey->priv_key,
444  NULL, NULL, ctx))
445  {
446  ECerr(EC_F_EC_KEY_CHECK_KEY, ERR_R_EC_LIB);
447  goto err;
448  }
449  if (EC_POINT_cmp(eckey->group, point, eckey->pub_key,
450  ctx) != 0)
451  {
452  ECerr(EC_F_EC_KEY_CHECK_KEY, EC_R_INVALID_PRIVATE_KEY);
453  goto err;
454  }
455  }
456  ok = 1;
457 err:
458  if (ctx != NULL)
459  BN_CTX_free(ctx);
460  if (point != NULL)
461  EC_POINT_free(point);
462  return(ok);
463  }
