73 void RC2_ecb_encrypt(const unsigned char *in, unsigned char *out, RC2_KEY *ks,
74  int encrypt)
75  {
76  unsigned long l,d[2];
77 
78  c2l(in,l); d[0]=l;
79  c2l(in,l); d[1]=l;
80  if (encrypt)
81  RC2_encrypt(d,ks);
82  else
83  RC2_decrypt(d,ks);
84  l=d[0]; l2c(l,out);
85  l=d[1]; l2c(l,out);
86  l=d[0]=d[1]=0;
87  }
