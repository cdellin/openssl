67 int HMAC_Init_ex(HMAC_CTX *ctx, const void *key, int len,
68  const EVP_MD *md, ENGINE *impl)
69  {
70  int i,j,reset=0;
71  unsigned char pad[HMAC_MAX_MD_CBLOCK];
72 
73  if (md != NULL)
74  {
75  reset=1;
76  ctx->md=md;
77  }
78  else
79  md=ctx->md;
80 
81  if (key != NULL)
82  {
83  reset=1;
84  j=M_EVP_MD_block_size(md);
85  OPENSSL_assert(j <= (int)sizeof(ctx->key));
86  if (j < len)
87  {
88  if (!EVP_DigestInit_ex(&ctx->md_ctx,md, impl))
89  goto err;
90  if (!EVP_DigestUpdate(&ctx->md_ctx,key,len))
91  goto err;
92  if (!EVP_DigestFinal_ex(&(ctx->md_ctx),ctx->key,
93  &ctx->key_length))
94  goto err;
95  }
96  else
97  {
98  OPENSSL_assert(len>=0 && len<=(int)sizeof(ctx->key));
99  memcpy(ctx->key,key,len);
100  ctx->key_length=len;
101  }
102  if(ctx->key_length != HMAC_MAX_MD_CBLOCK)
103  memset(&ctx->key[ctx->key_length], 0,
104  HMAC_MAX_MD_CBLOCK - ctx->key_length);
105  }
106 
107  if (reset)
108  {
109  for (i=0; i<HMAC_MAX_MD_CBLOCK; i++)
110  pad[i]=0x36^ctx->key[i];
111  if (!EVP_DigestInit_ex(&ctx->i_ctx,md, impl))
112  goto err;
113  if (!EVP_DigestUpdate(&ctx->i_ctx,pad,M_EVP_MD_block_size(md)))
114  goto err;
115 
116  for (i=0; i<HMAC_MAX_MD_CBLOCK; i++)
117  pad[i]=0x5c^ctx->key[i];
118  if (!EVP_DigestInit_ex(&ctx->o_ctx,md, impl))
119  goto err;
120  if (!EVP_DigestUpdate(&ctx->o_ctx,pad,M_EVP_MD_block_size(md)))
121  goto err;
122  }
123  if (!EVP_MD_CTX_copy_ex(&ctx->md_ctx,&ctx->i_ctx))
124  goto err;
125  return 1;
126  err:
127  return 0;
128  }

192 unsigned char *HMAC(const EVP_MD *evp_md, const void *key, int key_len,
193  const unsigned char *d, size_t n, unsigned char *md,
194  unsigned int *md_len)
195  {
196  HMAC_CTX c;
197  static unsigned char m[EVP_MAX_MD_SIZE];
198 
199  if (md == NULL) md=m;
200  HMAC_CTX_init(&c);
201  if (!HMAC_Init(&c,key,key_len,evp_md))
202  goto err;
203  if (!HMAC_Update(&c,d,n))
204  goto err;
205  if (!HMAC_Final(&c,md,md_len))
206  goto err;
207  HMAC_CTX_cleanup(&c);
208  return md;
209  err:
210  return NULL;
211  }
