65 int EVP_add_cipher(const EVP_CIPHER *c)
66  {
67  int r;
68 
69  if (c == NULL) return 0;
70 
71  r=OBJ_NAME_add(OBJ_nid2sn(c->nid),OBJ_NAME_TYPE_CIPHER_METH,(const char *)c);
72  if (r == 0) return(0);
73  check_defer(c->nid);
74  r=OBJ_NAME_add(OBJ_nid2ln(c->nid),OBJ_NAME_TYPE_CIPHER_METH,(const char *)c);
75  return(r);

79 int EVP_add_digest(const EVP_MD *md)
80  {
81  int r;
82  const char *name;
83 
84  name=OBJ_nid2sn(md->type);
85  r=OBJ_NAME_add(name,OBJ_NAME_TYPE_MD_METH,(const char *)md);
86  if (r == 0) return(0);
87  check_defer(md->type);
88  r=OBJ_NAME_add(OBJ_nid2ln(md->type),OBJ_NAME_TYPE_MD_METH,(const char *)md);
89  if (r == 0) return(0);
90 
91  if (md->pkey_type && md->type != md->pkey_type)
92  {
93  r=OBJ_NAME_add(OBJ_nid2sn(md->pkey_type),
94  OBJ_NAME_TYPE_MD_METH|OBJ_NAME_ALIAS,name);
95  if (r == 0) return(0);
96  check_defer(md->pkey_type);
97  r=OBJ_NAME_add(OBJ_nid2ln(md->pkey_type),
98  OBJ_NAME_TYPE_MD_METH|OBJ_NAME_ALIAS,name);
99  }
100  return(r);
101  }
