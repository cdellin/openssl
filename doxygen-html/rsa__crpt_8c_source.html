132 static BIGNUM *rsa_get_public_exp(const BIGNUM *d, const BIGNUM *p,
133  const BIGNUM *q, BN_CTX *ctx)
134 {
135  BIGNUM *ret = NULL, *r0, *r1, *r2;
136 
137  if (d == NULL || p == NULL || q == NULL)
138  return NULL;
139 
140  BN_CTX_start(ctx);
141  r0 = BN_CTX_get(ctx);
142  r1 = BN_CTX_get(ctx);
143  r2 = BN_CTX_get(ctx);
144  if (r2 == NULL)
145  goto err;
146 
147  if (!BN_sub(r1, p, BN_value_one())) goto err;
148  if (!BN_sub(r2, q, BN_value_one())) goto err;
149  if (!BN_mul(r0, r1, r2, ctx)) goto err;
150 
151  ret = BN_mod_inverse(NULL, d, r0, ctx);
152 err:
153  BN_CTX_end(ctx);
154  return ret;
155 }

157 BN_BLINDING *RSA_setup_blinding(RSA *rsa, BN_CTX *in_ctx)
158 {
159  BIGNUM local_n;
160  BIGNUM *e,*n;
161  BN_CTX *ctx;
162  BN_BLINDING *ret = NULL;
163 
164  if (in_ctx == NULL)
165  {
166  if ((ctx = BN_CTX_new()) == NULL) return 0;
167  }
168  else
169  ctx = in_ctx;
170 
171  BN_CTX_start(ctx);
172  e = BN_CTX_get(ctx);
173  if (e == NULL)
174  {
175  RSAerr(RSA_F_RSA_SETUP_BLINDING, ERR_R_MALLOC_FAILURE);
176  goto err;
177  }
178 
179  if (rsa->e == NULL)
180  {
181  e = rsa_get_public_exp(rsa->d, rsa->p, rsa->q, ctx);
182  if (e == NULL)
183  {
184  RSAerr(RSA_F_RSA_SETUP_BLINDING, RSA_R_NO_PUBLIC_EXPONENT);
185  goto err;
186  }
187  }
188  else
189  e = rsa->e;
190 
191 
192  if ((RAND_status() == 0) && rsa->d != NULL && rsa->d->d != NULL)
193  {
194  /* if PRNG is not properly seeded, resort to secret
195  * exponent as unpredictable seed */
196  RAND_add(rsa->d->d, rsa->d->dmax * sizeof rsa->d->d[0], 0.0);
197  }
198 
199  if (!(rsa->flags & RSA_FLAG_NO_CONSTTIME))
200  {
201  /* Set BN_FLG_CONSTTIME flag */
202  n = &local_n;
203  BN_with_flags(n, rsa->n, BN_FLG_CONSTTIME);
204  }
205  else
206  n = rsa->n;
207 
208  ret = BN_BLINDING_create_param(NULL, e, n, ctx,
209  rsa->meth->bn_mod_exp, rsa->_method_mod_n);
210  if (ret == NULL)
211  {
212  RSAerr(RSA_F_RSA_SETUP_BLINDING, ERR_R_BN_LIB);
213  goto err;
214  }
215  CRYPTO_THREADID_current(BN_BLINDING_thread_id(ret));
216 err:
217  BN_CTX_end(ctx);
218  if (in_ctx == NULL)
219  BN_CTX_free(ctx);
220  if(rsa->e == NULL)
221  BN_free(e);
222 
223  return ret;
224 }
