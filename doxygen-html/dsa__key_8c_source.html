102 static int dsa_builtin_keygen(DSA *dsa)
103  {
104  int ok=0;
105  BN_CTX *ctx=NULL;
106  BIGNUM *pub_key=NULL,*priv_key=NULL;
107 
108 #ifdef OPENSSL_FIPS
109  if (FIPS_module_mode() && !(dsa->flags & DSA_FLAG_NON_FIPS_ALLOW)
110  && (BN_num_bits(dsa->p) < OPENSSL_DSA_FIPS_MIN_MODULUS_BITS))
111  {
112  DSAerr(DSA_F_DSA_BUILTIN_KEYGEN, DSA_R_KEY_SIZE_TOO_SMALL);
113  goto err;
114  }
115  if (!fips_check_dsa_prng(dsa, 0, 0))
116  goto err;
117 #endif
118 
119  if ((ctx=BN_CTX_new()) == NULL) goto err;
120 
121  if (dsa->priv_key == NULL)
122  {
123  if ((priv_key=BN_new()) == NULL) goto err;
124  }
125  else
126  priv_key=dsa->priv_key;
127 
128  do
129  if (!BN_rand_range(priv_key,dsa->q)) goto err;
130  while (BN_is_zero(priv_key));
131 
132  if (dsa->pub_key == NULL)
133  {
134  if ((pub_key=BN_new()) == NULL) goto err;
135  }
136  else
137  pub_key=dsa->pub_key;
138 
139  {
140  BIGNUM local_prk;
141  BIGNUM *prk;
142 
143  if ((dsa->flags & DSA_FLAG_NO_EXP_CONSTTIME) == 0)
144  {
145  BN_init(&local_prk);
146  prk = &local_prk;
147  BN_with_flags(prk, priv_key, BN_FLG_CONSTTIME);
148  }
149  else
150  prk = priv_key;
151 
152  if (!BN_mod_exp(pub_key,dsa->g,prk,dsa->p,ctx)) goto err;
153  }
154 
155  dsa->priv_key=priv_key;
156  dsa->pub_key=pub_key;
157 #ifdef OPENSSL_FIPS
158  if(!fips_check_dsa(dsa))
159  {
160  dsa->pub_key = NULL;
161  dsa->priv_key = NULL;
162  goto err;
163  }
164 #endif
165  ok=1;
166 
167 err:
168  if ((pub_key != NULL) && (dsa->pub_key == NULL)) BN_free(pub_key);
169  if ((priv_key != NULL) && (dsa->priv_key == NULL)) BN_free(priv_key);
170  if (ctx != NULL) BN_CTX_free(ctx);
171  return(ok);
172  }
