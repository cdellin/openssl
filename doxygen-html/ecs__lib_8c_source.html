180 ECDSA_DATA *ecdsa_check(EC_KEY *key)
181 {
182  ECDSA_DATA *ecdsa_data;
183 
184  void *data = EC_KEY_get_key_method_data(key, ecdsa_data_dup,
185  ecdsa_data_free, ecdsa_data_free);
186  if (data == NULL)
187  {
188  ecdsa_data = (ECDSA_DATA *)ecdsa_data_new();
189  if (ecdsa_data == NULL)
190  return NULL;
191  data = EC_KEY_insert_key_method_data(key, (void *)ecdsa_data,
192  ecdsa_data_dup, ecdsa_data_free, ecdsa_data_free);
193  if (data != NULL)
194  {
195  /* Another thread raced us to install the key_method
196  * data and won. */
197  ecdsa_data_free(ecdsa_data);
198  ecdsa_data = (ECDSA_DATA *)data;
199  }
200  }
201  else
202  ecdsa_data = (ECDSA_DATA *)data;
203 
204 
205  return ecdsa_data;
206 }

208 int ECDSA_size(const EC_KEY *r)
209 {
210  int ret,i;
211  ASN1_INTEGER bs;
212  BIGNUM *order=NULL;
213  unsigned char buf[4];
214  const EC_GROUP *group;
215 
216  if (r == NULL)
217  return 0;
218  group = EC_KEY_get0_group(r);
219  if (group == NULL)
220  return 0;
221 
222  if ((order = BN_new()) == NULL) return 0;
223  if (!EC_GROUP_get_order(group,order,NULL))
224  {
225  BN_clear_free(order);
226  return 0;
227  }
228  i=BN_num_bits(order);
229  bs.length=(i+7)/8;
230  bs.data=buf;
231  bs.type=V_ASN1_INTEGER;
232  /* If the top bit is set the asn1 encoding is 1 larger. */
233  buf[0]=0xff;
234 
235  i=i2d_ASN1_INTEGER(&bs,NULL);
236  i+=i; /* r and s */
237  ret=ASN1_object_size(1,i,V_ASN1_SEQUENCE);
238  BN_clear_free(order);
239  return(ret);
240 }
