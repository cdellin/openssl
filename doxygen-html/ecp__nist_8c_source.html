123 int ec_GFp_nist_group_set_curve(EC_GROUP *group, const BIGNUM *p,
124  const BIGNUM *a, const BIGNUM *b, BN_CTX *ctx)
125  {
126  int ret = 0;
127  BN_CTX *new_ctx = NULL;
128  BIGNUM *tmp_bn;
129 
130  if (ctx == NULL)
131  if ((ctx = new_ctx = BN_CTX_new()) == NULL) return 0;
132 
133  BN_CTX_start(ctx);
134  if ((tmp_bn = BN_CTX_get(ctx)) == NULL) goto err;
135 
136  if (BN_ucmp(BN_get0_nist_prime_192(), p) == 0)
137  group->field_mod_func = BN_nist_mod_192;
138  else if (BN_ucmp(BN_get0_nist_prime_224(), p) == 0)
139  group->field_mod_func = BN_nist_mod_224;
140  else if (BN_ucmp(BN_get0_nist_prime_256(), p) == 0)
141  group->field_mod_func = BN_nist_mod_256;
142  else if (BN_ucmp(BN_get0_nist_prime_384(), p) == 0)
143  group->field_mod_func = BN_nist_mod_384;
144  else if (BN_ucmp(BN_get0_nist_prime_521(), p) == 0)
145  group->field_mod_func = BN_nist_mod_521;
146  else
147  {
148  ECerr(EC_F_EC_GFP_NIST_GROUP_SET_CURVE, EC_R_NOT_A_NIST_PRIME);
149  goto err;
150  }
151 
152  ret = ec_GFp_simple_group_set_curve(group, p, a, b, ctx);
153 
154  err:
155  BN_CTX_end(ctx);
156  if (new_ctx != NULL)
157  BN_CTX_free(new_ctx);
158  return ret;
