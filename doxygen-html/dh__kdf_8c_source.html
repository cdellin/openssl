67 static int skip_asn1(unsigned char **pp, long *plen, int exptag)
68  {
69  const unsigned char *q = *pp;
70  int i, tag, xclass;
71  long tmplen;
72  i = ASN1_get_object(&q, &tmplen, &tag, &xclass, *plen);
73  if (i & 0x80)
74  return 0;
75  if (tag != exptag || xclass != V_ASN1_UNIVERSAL)
76  return 0;
77  if (tag == V_ASN1_OBJECT)
78  q += tmplen;
79  *plen -= q - *pp;
80  *pp = (unsigned char *)q;
81  return 1;
82  }

89 static int dh_sharedinfo_encode(unsigned char **pder, unsigned char **pctr,
90  ASN1_OBJECT *key_oid, size_t outlen,
91  const unsigned char *ukm, size_t ukmlen)
92  {
93  unsigned char *p;
94  int derlen;
95  long tlen;
96  /* "magic" value to check offset is sane */
97  static unsigned char ctr[4] = {0xF3, 0x17, 0x22, 0x53};
98  X509_ALGOR atmp;
99  ASN1_OCTET_STRING ctr_oct, ukm_oct, *pukm_oct;
100  ASN1_TYPE ctr_atype;
101  if (ukmlen > DH_KDF_MAX || outlen > DH_KDF_MAX)
102  return 0;
103  ctr_oct.data = ctr;
104  ctr_oct.length = 4;
105  ctr_oct.flags = 0;
106  ctr_oct.type = V_ASN1_OCTET_STRING;
107  ctr_atype.type = V_ASN1_OCTET_STRING;
108  ctr_atype.value.octet_string = &ctr_oct;
109  atmp.algorithm = key_oid;
110  atmp.parameter = &ctr_atype;
111  if (ukm)
112  {
113  ukm_oct.type = V_ASN1_OCTET_STRING;
114  ukm_oct.flags = 0;
115  ukm_oct.data = (unsigned char *)ukm;
116  ukm_oct.length = ukmlen;
117  pukm_oct = &ukm_oct;
118  }
119  else
120  pukm_oct = NULL;
121  derlen = CMS_SharedInfo_encode(pder, &atmp, pukm_oct, outlen);
122  if (derlen <= 0)
123  return 0;
124  p = *pder;
125  tlen = derlen;
126  if (!skip_asn1(&p, &tlen, V_ASN1_SEQUENCE))
127  return 0;
128  if (!skip_asn1(&p, &tlen, V_ASN1_SEQUENCE))
129  return 0;
130  if (!skip_asn1(&p, &tlen, V_ASN1_OBJECT))
131  return 0;
132  if (!skip_asn1(&p, &tlen, V_ASN1_OCTET_STRING))
133  return 0;
134  if (memcmp(p, ctr, 4))
135  return 0;
136  *pctr = p;
137  return derlen;
138  }

140 int DH_KDF_X9_42(unsigned char *out, size_t outlen,
141  const unsigned char *Z, size_t Zlen,
142  ASN1_OBJECT *key_oid,
143  const unsigned char *ukm, size_t ukmlen,
144  const EVP_MD *md)
145  {
146  EVP_MD_CTX mctx;
147  int rv = 0;
148  unsigned int i;
149  size_t mdlen;
150  unsigned char *der = NULL, *ctr;
151  int derlen;
152  if (Zlen > DH_KDF_MAX)
153  return 0;
154  mdlen = EVP_MD_size(md);
155  EVP_MD_CTX_init(&mctx);
156  derlen = dh_sharedinfo_encode(&der, &ctr, key_oid, outlen,
157  ukm, ukmlen);
158  if (derlen == 0)
159  goto err;
160  for (i = 1;;i++)
161  {
162  unsigned char mtmp[EVP_MAX_MD_SIZE];
163  EVP_DigestInit_ex(&mctx, md, NULL);
164  if (!EVP_DigestUpdate(&mctx, Z, Zlen))
165  goto err;
166  ctr[3] = i & 0xFF;
167  ctr[2] = (i >> 8) & 0xFF;
168  ctr[1] = (i >> 16) & 0xFF;
169  ctr[0] = (i >> 24) & 0xFF;
170  if (!EVP_DigestUpdate(&mctx, der, derlen))
171  goto err;
172  if (outlen >= mdlen)
173  {
174  if (!EVP_DigestFinal(&mctx, out, NULL))
175  goto err;
176  outlen -= mdlen;
177  if (outlen == 0)
178  break;
179  out += mdlen;
180  }
181  else
182  {
183  if (!EVP_DigestFinal(&mctx, mtmp, NULL))
184  goto err;
185  memcpy(out, mtmp, outlen);
186  OPENSSL_cleanse(mtmp, mdlen);
187  break;
188  }
189  }
190  rv = 1;
191  err:
192  if (der)
193  OPENSSL_free(der);
194  EVP_MD_CTX_cleanup(&mctx);
195  return rv;
196  }
