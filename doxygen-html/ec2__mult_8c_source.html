79 /* Compute the x-coordinate x/z for the point 2*(x/z) in Montgomery projective
80  * coordinates.
81  * Uses algorithm Mdouble in appendix of
82  * Lopez, J. and Dahab, R. "Fast multiplication on elliptic curves over
83  * GF(2^m) without precomputation" (CHES '99, LNCS 1717).
84  * modified to not require precomputation of c=b^{2^{m-1}}.
85  */
86 static int gf2m_Mdouble(const EC_GROUP *group, BIGNUM *x, BIGNUM *z, BN_CTX *ctx)
87  {
88  BIGNUM *t1;
89  int ret = 0;
90 
91  /* Since Mdouble is static we can guarantee that ctx != NULL. */
92  BN_CTX_start(ctx);
93  t1 = BN_CTX_get(ctx);
94  if (t1 == NULL) goto err;
95 
96  if (!group->meth->field_sqr(group, x, x, ctx)) goto err;
97  if (!group->meth->field_sqr(group, t1, z, ctx)) goto err;
98  if (!group->meth->field_mul(group, z, x, t1, ctx)) goto err;
99  if (!group->meth->field_sqr(group, x, x, ctx)) goto err;
100  if (!group->meth->field_sqr(group, t1, t1, ctx)) goto err;
101  if (!group->meth->field_mul(group, t1, &group->b, t1, ctx)) goto err;
102  if (!BN_GF2m_add(x, x, t1)) goto err;
103 
104  ret = 1;
105 
106  err:
107  BN_CTX_end(ctx);
108  return ret;
109  }

112  * projective coordinates.
113  * Uses algorithm Madd in appendix of
114  * Lopez, J. and Dahab, R. "Fast multiplication on elliptic curves over
115  * GF(2^m) without precomputation" (CHES '99, LNCS 1717).

118  const BIGNUM *x2, const BIGNUM *z2, BN_CTX *ctx)

122

124  BN_CTX_start(ctx);

127  if (t2 == NULL) goto err;
128 
129  if (!BN_copy(t1, x)) goto err;

145 /* Compute the x, y affine coordinates from the point (x1, z1) (x2, z2)
146  * using Montgomery point multiplication algorithm Mxy() in appendix of
147  * Lopez, J. and Dahab, R. "Fast multiplication on elliptic curves over
148  * GF(2^m) without precomputation" (CHES '99, LNCS 1717).
149  * Returns:
150  * 0 on error
151  * 1 if return value should be the point at infinity
152  * 2 otherwise
153  */
154 static int gf2m_Mxy(const EC_GROUP *group, const BIGNUM *x, const BIGNUM *y, BIGNUM *x1,
155  BIGNUM *z1, BIGNUM *x2, BIGNUM *z2, BN_CTX *ctx)
156  {
157  BIGNUM *t3, *t4, *t5;
158  int ret = 0;
159 
160  if (BN_is_zero(z1))
161  {
162  BN_zero(x2);
163  BN_zero(z2);
164  return 1;
165  }
166 
167  if (BN_is_zero(z2))
168  {
169  if (!BN_copy(x2, x)) return 0;
170  if (!BN_GF2m_add(z2, x, y)) return 0;
171  return 2;
172  }
173 
174  /* Since Mxy is static we can guarantee that ctx != NULL. */
175  BN_CTX_start(ctx);
176  t3 = BN_CTX_get(ctx);
177  t4 = BN_CTX_get(ctx);
178  t5 = BN_CTX_get(ctx);
179  if (t5 == NULL) goto err;
180 
181  if (!BN_one(t5)) goto err;
182 
183  if (!group->meth->field_mul(group, t3, z1, z2, ctx)) goto err;
184 
185  if (!group->meth->field_mul(group, z1, z1, x, ctx)) goto err;
186  if (!BN_GF2m_add(z1, z1, x1)) goto err;
187  if (!group->meth->field_mul(group, z2, z2, x, ctx)) goto err;
188  if (!group->meth->field_mul(group, x1, z2, x1, ctx)) goto err;
189  if (!BN_GF2m_add(z2, z2, x2)) goto err;
190 
191  if (!group->meth->field_mul(group, z2, z2, z1, ctx)) goto err;
192  if (!group->meth->field_sqr(group, t4, x, ctx)) goto err;
193  if (!BN_GF2m_add(t4, t4, y)) goto err;
194  if (!group->meth->field_mul(group, t4, t4, t3, ctx)) goto err;
195  if (!BN_GF2m_add(t4, t4, z2)) goto err;
196 
197  if (!group->meth->field_mul(group, t3, t3, x, ctx)) goto err;
198  if (!group->meth->field_div(group, t3, t5, t3, ctx)) goto err;
199  if (!group->meth->field_mul(group, t4, t3, t4, ctx)) goto err;
200  if (!group->meth->field_mul(group, x2, x1, t3, ctx)) goto err;
201  if (!BN_GF2m_add(z2, x2, x)) goto err;
202 
203  if (!group->meth->field_mul(group, z2, z2, t4, ctx)) goto err;
204  if (!BN_GF2m_add(z2, z2, y)) goto err;
205 
206  ret = 2;
207 
208  err:
209  BN_CTX_end(ctx);
210  return ret;
211  }

214 /* Computes scalar*point and stores the result in r.
215  * point can not equal r.
216  * Uses a modified algorithm 2P of
217  * Lopez, J. and Dahab, R. "Fast multiplication on elliptic curves over
218  * GF(2^m) without precomputation" (CHES '99, LNCS 1717).
219  *
220  * To protect against side-channel attack the function uses constant time swap,
221  * avoiding conditional branches.
222  */
223 static int ec_GF2m_montgomery_point_multiply(const EC_GROUP *group, EC_POINT *r, const BIGNUM *scalar,
224  const EC_POINT *point, BN_CTX *ctx)
225  {
226  BIGNUM *x1, *x2, *z1, *z2;
227  int ret = 0, i;
228  BN_ULONG mask,word;
229 
230  if (r == point)
231  {
232  ECerr(EC_F_EC_GF2M_MONTGOMERY_POINT_MULTIPLY, EC_R_INVALID_ARGUMENT);
233  return 0;
234  }
235 
236  /* if result should be point at infinity */
237  if ((scalar == NULL) || BN_is_zero(scalar) || (point == NULL) ||
238  EC_POINT_is_at_infinity(group, point))
239  {
240  return EC_POINT_set_to_infinity(group, r);
241  }
242 
243  /* only support affine coordinates */
244  if (!point->Z_is_one) return 0;
245 
246  /* Since point_multiply is static we can guarantee that ctx != NULL. */
247  BN_CTX_start(ctx);
248  x1 = BN_CTX_get(ctx);
249  z1 = BN_CTX_get(ctx);
250  if (z1 == NULL) goto err;
251 
252  x2 = &r->X;
253  z2 = &r->Y;
254 
255  bn_wexpand(x1, group->field.top);
256  bn_wexpand(z1, group->field.top);
257  bn_wexpand(x2, group->field.top);
258  bn_wexpand(z2, group->field.top);
259 
260  if (!BN_GF2m_mod_arr(x1, &point->X, group->poly)) goto err; /* x1 = x */
261  if (!BN_one(z1)) goto err; /* z1 = 1 */
262  if (!group->meth->field_sqr(group, z2, x1, ctx)) goto err; /* z2 = x1^2 = x^2 */
263  if (!group->meth->field_sqr(group, x2, z2, ctx)) goto err;
264  if (!BN_GF2m_add(x2, x2, &group->b)) goto err; /* x2 = x^4 + b */
265 
266  /* find top most bit and go one past it */
267  i = scalar->top - 1;
268  mask = BN_TBIT;
269  word = scalar->d[i];
270  while (!(word & mask)) mask >>= 1;
271  mask >>= 1;
272  /* if top most bit was at word break, go to next word */
273  if (!mask)
274  {
275  i--;
276  mask = BN_TBIT;
277  }
278 
279  for (; i >= 0; i--)
280  {
281  word = scalar->d[i];
282  while (mask)
283  {
284  BN_consttime_swap(word & mask, x1, x2, group->field.top);
285  BN_consttime_swap(word & mask, z1, z2, group->field.top);
286  if (!gf2m_Madd(group, &point->X, x2, z2, x1, z1, ctx)) goto err;
287  if (!gf2m_Mdouble(group, x1, z1, ctx)) goto err;
288  BN_consttime_swap(word & mask, x1, x2, group->field.top);
289  BN_consttime_swap(word & mask, z1, z2, group->field.top);
290  mask >>= 1;
291  }
292  mask = BN_TBIT;
293  }
294 
295  /* convert out of "projective" coordinates */
296  i = gf2m_Mxy(group, &point->X, &point->Y, x1, z1, x2, z2, ctx);
297  if (i == 0) goto err;
298  else if (i == 1)
299  {
300  if (!EC_POINT_set_to_infinity(group, r)) goto err;
301  }
302  else
303  {
304  if (!BN_one(&r->Z)) goto err;
305  r->Z_is_one = 1;
306  }
307 
308  /* GF(2^m) field elements should always have BIGNUM::neg = 0 */
309  BN_set_negative(&r->X, 0);
310  BN_set_negative(&r->Y, 0);
311 
312  ret = 1;
313 
314  err:
315  BN_CTX_end(ctx);
316  return ret;
317  }

320 /* Computes the sum
321  * scalar*group->generator + scalars[0]*points[0] + ... + scalars[num-1]*points[num-1]
322  * gracefully ignoring NULL scalar values.
323  */
324 int ec_GF2m_simple_mul(const EC_GROUP *group, EC_POINT *r, const BIGNUM *scalar,
325  size_t num, const EC_POINT *points[], const BIGNUM *scalars[], BN_CTX *ctx)
326  {
327  BN_CTX *new_ctx = NULL;
328  int ret = 0;
329  size_t i;
330  EC_POINT *p=NULL;
331  EC_POINT *acc = NULL;
332 
333  if (ctx == NULL)
334  {
335  ctx = new_ctx = BN_CTX_new();
336  if (ctx == NULL)
337  return 0;
338  }
339 
340  /* This implementation is more efficient than the wNAF implementation for 2
341  * or fewer points. Use the ec_wNAF_mul implementation for 3 or more points,
342  * or if we can perform a fast multiplication based on precomputation.
343  */
344  if ((scalar && (num > 1)) || (num > 2) || (num == 0 && EC_GROUP_have_precompute_mult(group)))
345  {
346  ret = ec_wNAF_mul(group, r, scalar, num, points, scalars, ctx);
347  goto err;
348  }
349 
350  if ((p = EC_POINT_new(group)) == NULL) goto err;
351  if ((acc = EC_POINT_new(group)) == NULL) goto err;
352 
353  if (!EC_POINT_set_to_infinity(group, acc)) goto err;
354 
355  if (scalar)
356  {
357  if (!ec_GF2m_montgomery_point_multiply(group, p, scalar, group->generator, ctx)) goto err;
358  if (BN_is_negative(scalar))
359  if (!group->meth->invert(group, p, ctx)) goto err;
360  if (!group->meth->add(group, acc, acc, p, ctx)) goto err;
361  }
362 
363  for (i = 0; i < num; i++)
364  {
365  if (!ec_GF2m_montgomery_point_multiply(group, p, scalars[i], points[i], ctx)) goto err;
366  if (BN_is_negative(scalars[i]))
367  if (!group->meth->invert(group, p, ctx)) goto err;
368  if (!group->meth->add(group, acc, acc, p, ctx)) goto err;
369  }
370 
371  if (!EC_POINT_copy(r, acc)) goto err;
372 
373  ret = 1;
374 
375  err:
376  if (p) EC_POINT_free(p);
377  if (acc) EC_POINT_free(acc);
378  if (new_ctx != NULL)
379  BN_CTX_free(new_ctx);
380  return ret;
381  }
