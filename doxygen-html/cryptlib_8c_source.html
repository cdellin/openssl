141  IA32CAP vec;

143 
144  if (trigger) return;

146  trigger=1;
147  if ((env=getenv("OPENSSL_ia32cap"))) {
148  int off = (env[0]=='~')?1:0;
149 #if defined(_WIN32)
150  if (!sscanf(env+off,"%I64i",&vec)) vec = strtoul(env+off,NULL,0);
151 #else
152  if (!sscanf(env+off,"%lli",(long long *)&vec)) vec = strtoul(env+off,NULL,0);
153 #endif
154  if (off) vec = OPENSSL_ia32_cpuid(OPENSSL_ia32cap_P)&~vec;
155  else if (env[0]==':') vec = OPENSSL_ia32_cpuid(OPENSSL_ia32cap_P);
156 
157  OPENSSL_ia32cap_P[2] = 0;
158  if ((env=strchr(env,':'))) {
159  unsigned int vecx;
160  env++;
161  off = (env[0]=='~')?1:0;
162  vecx = strtoul(env+off,NULL,0);

164  else OPENSSL_ia32cap_P[2] = vecx;
