132 int BN_div_recp(BIGNUM *dv, BIGNUM *rem, const BIGNUM *m,
133  BN_RECP_CTX *recp, BN_CTX *ctx)
134  {
135  int i,j,ret=0;
136  BIGNUM *a,*b,*d,*r;
137 
138  BN_CTX_start(ctx);
139  a=BN_CTX_get(ctx);
140  b=BN_CTX_get(ctx);
141  if (dv != NULL)
142  d=dv;
143  else
144  d=BN_CTX_get(ctx);
145  if (rem != NULL)
146  r=rem;
147  else
148  r=BN_CTX_get(ctx);
149  if (a == NULL || b == NULL || d == NULL || r == NULL) goto err;
150 
151  if (BN_ucmp(m,&(recp->N)) < 0)
152  {
153  BN_zero(d);
154  if (!BN_copy(r,m)) return 0;
155  BN_CTX_end(ctx);
156  return(1);
157  }
158 
159  /* We want the remainder
160  * Given input of ABCDEF / ab
161  * we need multiply ABCDEF by 3 digests of the reciprocal of ab
162  *
163  */
164 
165  /* i := max(BN_num_bits(m), 2*BN_num_bits(N)) */
166  i=BN_num_bits(m);
167  j=recp->num_bits<<1;
168  if (j>i) i=j;
169 
170  /* Nr := round(2^i / N) */
171  if (i != recp->shift)
172  recp->shift=BN_reciprocal(&(recp->Nr),&(recp->N),
173  i,ctx); /* BN_reciprocal returns i, or -1 for an error */
174  if (recp->shift == -1) goto err;
175 
176  /* d := |round(round(m / 2^BN_num_bits(N)) * recp->Nr / 2^(i - BN_num_bits(N)))|
177  * = |round(round(m / 2^BN_num_bits(N)) * round(2^i / N) / 2^(i - BN_num_bits(N)))|
178  * <= |(m / 2^BN_num_bits(N)) * (2^i / N) * (2^BN_num_bits(N) / 2^i)|
179  * = |m/N|
180  */
181  if (!BN_rshift(a,m,recp->num_bits)) goto err;
182  if (!BN_mul(b,a,&(recp->Nr),ctx)) goto err;
183  if (!BN_rshift(d,b,i-recp->num_bits)) goto err;
184  d->neg=0;
185 
186  if (!BN_mul(b,&(recp->N),d,ctx)) goto err;
187  if (!BN_usub(r,m,b)) goto err;
188  r->neg=0;
189 
190 #if 1
191  j=0;
192  while (BN_ucmp(r,&(recp->N)) >= 0)
193  {
194  if (j++ > 2)
195  {
196  BNerr(BN_F_BN_DIV_RECP,BN_R_BAD_RECIPROCAL);
197  goto err;
198  }
199  if (!BN_usub(r,r,&(recp->N))) goto err;
200  if (!BN_add_word(d,1)) goto err;
201  }
202 #endif
203 
204  r->neg=BN_is_zero(r)?0:m->neg;
205  d->neg=m->neg^recp->N.neg;
206  ret=1;
207 err:
208  BN_CTX_end(ctx);
209  bn_check_top(dv);
210  bn_check_top(rem);
211  return(ret);
212  }
