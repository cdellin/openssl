423 /* number of Miller-Rabin iterations for an error rate of less than 2^-80
424  * for random 'b'-bit input, b >= 100 (taken from table 4.4 in the Handbook
425  * of Applied Cryptography [Menezes, van Oorschot, Vanstone; CRC Press 1996];
426  * original paper: Damgaard, Landrock, Pomerance: Average case error estimates
427  * for the strong probable prime test. -- Math. Comp. 61 (1993) 177-194) */
428 #define BN_prime_checks_for_size(b) ((b) >= 1300 ? 2 : \
429  (b) >= 850 ? 3 : \
430  (b) >= 650 ? 4 : \
431  (b) >= 550 ? 5 : \
432  (b) >= 450 ? 6 : \
433  (b) >= 400 ? 7 : \
434  (b) >= 350 ? 8 : \
435  (b) >= 300 ? 9 : \
436  (b) >= 250 ? 12 : \
437  (b) >= 200 ? 15 : \
438  (b) >= 150 ? 18 : \
439  /* b >= 100 */ 27)

753 /* Bignum consistency macros
754  * There is one "API" macro, bn_fix_top(), for stripping leading zeroes from
755  * bignum data after direct manipulations on the data. There is also an
756  * "internal" macro, bn_check_top(), for verifying that there are no leading
757  * zeroes. Unfortunately, some auditing is required due to the fact that
758  * bn_fix_top() has become an overabused duct-tape because bignum data is
759  * occasionally passed around in an inconsistent state. So the following
760  * changes have been made to sort this out;
761  * - bn_fix_top()s implementation has been moved to bn_correct_top()
762  * - if BN_DEBUG isn't defined, bn_fix_top() maps to bn_correct_top(), and
763  * bn_check_top() is as before.
764  * - if BN_DEBUG *is* defined;
765  * - bn_check_top() tries to pollute unused words even if the bignum 'top' is
766  * consistent. (ed: only if BN_DEBUG_RAND is defined)
767  * - bn_fix_top() maps to bn_check_top() rather than "fixing" anything.
768  * The idea is to have debug builds flag up inconsistent bignums when they
769  * occur. If that occurs in a bn_fix_top(), we examine the code in question; if
770  * the use of bn_fix_top() was appropriate (ie. it follows directly after code
771  * that manipulates the bignum) it is converted to bn_correct_top(), and if it
772  * was not appropriate, we convert it permanently to bn_check_top() and track
773  * down the cause of the bug. Eventually, no internal code should be using the
774  * bn_fix_top() macro. External applications and libraries should try this with
775  * their own code too, both in terms of building against the openssl headers
776  * with BN_DEBUG defined *and* linking with a version of OpenSSL built with it
777  * defined. This not only improves external code, it provides more test
778  * coverage for openssl's own code.
779  */
