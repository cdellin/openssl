70 int TS_ASN1_INTEGER_print_bio(BIO *bio, const ASN1_INTEGER *num)
71  {
72  BIGNUM num_bn;
73  int result = 0;
74  char *hex;
75 
76  BN_init(&num_bn);
77  ASN1_INTEGER_to_BN(num, &num_bn);
78  if ((hex = BN_bn2hex(&num_bn)))
79  {
80  result = BIO_write(bio, "0x", 2) > 0;
81  result = result && BIO_write(bio, hex, strlen(hex)) > 0;
82  OPENSSL_free(hex);
83  }
84  BN_free(&num_bn);
85 
86  return result;
87  }
