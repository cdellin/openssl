76  if(pbe_nid == -1)
77  pbe = PKCS5_pbe2_set(cipher, iter, salt, saltlen);

79  pbe = PKCS5_pbe2_set_iv(cipher, iter, salt, saltlen, NULL, pbe_nid);
80  else
81  {

84  }
85  if(!pbe) {

87  goto err;
88  }
89  X509_ALGOR_free(p8->algor);
90  p8->algor = pbe;
91  M_ASN1_OCTET_STRING_free(p8->digest);
92  p8->digest = PKCS12_item_i2d_encrypt(pbe, ASN1_ITEM_rptr(PKCS8_PRIV_KEY_INFO),
93  pass, passlen, p8inf, 1);
94  if(!p8->digest) {
95  PKCS12err(PKCS12_F_PKCS8_ENCRYPT, PKCS12_R_ENCRYPT_ERROR);
96  goto err;
97  }
98 
99  return p8;
