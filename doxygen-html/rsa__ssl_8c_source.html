67 int RSA_padding_add_SSLv23(unsigned char *to, int tlen,
68  const unsigned char *from, int flen)
69  {
70  int i,j;
71  unsigned char *p;
72 
73  if (flen > (tlen-11))
74  {
75  RSAerr(RSA_F_RSA_PADDING_ADD_SSLV23,RSA_R_DATA_TOO_LARGE_FOR_KEY_SIZE);
76  return(0);
77  }
78 
79  p=(unsigned char *)to;
80 
81  *(p++)=0;
82  *(p++)=2; /* Public Key BT (Block Type) */
83 
84  /* pad out with non-zero random data */
85  j=tlen-3-8-flen;
86 
87  if (RAND_bytes(p,j) <= 0)
88  return(0);
89  for (i=0; i<j; i++)
90  {
91  if (*p == '\0')
92  do {
93  if (RAND_bytes(p,1) <= 0)
94  return(0);
95  } while (*p == '\0');
96  p++;
97  }
98 
99  memset(p,3,8);
100  p+=8;
101  *(p++)='\0';
102 
103  memcpy(p,from,(unsigned int)flen);
104  return(1);
105  }

107 int RSA_padding_check_SSLv23(unsigned char *to, int tlen,
108  const unsigned char *from, int flen, int num)
109  {
110  int i,j,k;
111  const unsigned char *p;
112 
113  p=from;
114  if (flen < 10)
115  {
116  RSAerr(RSA_F_RSA_PADDING_CHECK_SSLV23,RSA_R_DATA_TOO_SMALL);
117  return(-1);
118  }
119  if ((num != (flen+1)) || (*(p++) != 02))
120  {
121  RSAerr(RSA_F_RSA_PADDING_CHECK_SSLV23,RSA_R_BLOCK_TYPE_IS_NOT_02);
122  return(-1);
123  }
124 
125  /* scan over padding data */
126  j=flen-1; /* one for type */
127  for (i=0; i<j; i++)
128  if (*(p++) == 0) break;
129 
130  if ((i == j) || (i < 8))
131  {
132  RSAerr(RSA_F_RSA_PADDING_CHECK_SSLV23,RSA_R_NULL_BEFORE_BLOCK_MISSING);
133  return(-1);
134  }
135  for (k = -9; k<-1; k++)
136  {
137  if (p[k] != 0x03) break;
138  }
139  if (k == -1)
140  {
141  RSAerr(RSA_F_RSA_PADDING_CHECK_SSLV23,RSA_R_SSLV3_ROLLBACK_ATTACK);
142  return(-1);
143  }
144 
145  i++; /* Skip over the '\0' */
146  j-=i;
147  if (j > tlen)
148  {
149  RSAerr(RSA_F_RSA_PADDING_CHECK_SSLV23,RSA_R_DATA_TOO_LARGE);
150  return(-1);
151  }
152  memcpy(to,p,(unsigned int)j);
153 
154  return(j);
155  }
